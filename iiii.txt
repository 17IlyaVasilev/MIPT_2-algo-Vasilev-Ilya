#include <iostream>
#include <vector>

long long find_sum_mdiv_composite(const int& n) {
    std::vector<int> primes;
    std::vector<int> mind;
    for (int i = 0; i <= n; ++i) {
        mind.push_back(i);
    }
    for (int k = 2; k <= n; ++k) {
        if (mind[k] == k) {
            primes.push_back(k);
        }
        for (int i = 0; i < primes.size(); ++i) {
            if (primes[i] * k <= n && primes[i] <= mind[k]) {
                mind[primes[i] * k] = primes[i];
            }
            else {
                break;
            }
        }
    }

    long long ans = 0;
    for (int i = 4; i <= n; ++i) {
        if (mind[i] != i) {
            ans += mind[i];
        }
    }
    return ans;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);

    int n;
    std::cin >> n;

    std::cout << find_sum_mdiv_composite(n);

    return 0;
}
