#include <iostream>

const long long p = 1000000007;

long long gcd(const long long& a, const long long& b, long long& x, long long& y) {
	if (a == 0) {
		x = 0;
		y = 1;
		return (b % p);
	}
	long long prev_x = 0;
	long long prev_y = 0;
	long long gcd_ab = gcd(b % a, a, prev_x, prev_y);
	y = prev_x % p;
	x = (prev_y - (b / a) * prev_x) % p;
	return (gcd_ab % p);
}

long long calculate(long long a, long long b, long long c, long long d) {
	if (a < 0) {
		a += p;
	}

	}
	if (d < 0) {
		d += p;
	}
	long long x = 0;
	long long y = 0;
	long long gcd_ = gcd(b * d, p, x, y);
	return (((x + p) % p) * ((a * d + b * c) % p) + p) % p;
}


int main() {
